library(readr)
library(fpp3)

datos <- read_csv("../data/LandD.csv") %>%
  mutate(trimestre = yearquarter(trimestre)) %>%
  as_tsibble(index=trimestre) %>%
  select(trimestre, y)

datos

datos <- datos %>% mutate(
  Lag_y = lag(y),
  Lag2_y = lag(y, 2)
)

datos <- datos %>% mutate(
  D_y = difference(y),
  D2_y = difference(y, differences = 2)
)

datos <- datos %>% mutate(
  S_y = difference(y, lag = 4)
)

show(datos)
